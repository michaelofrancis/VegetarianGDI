my_skim <- skim_with(
    numeric = sfl(
        p25 = NULL,
        p50=NULL,
        p75=NULL,
        median = ~median(.x, na.rm=TRUE),
        min = ~min(.x, na.rm=TRUE), 
        max = ~max(.x, na.rm=TRUE), 
        `Data completeness (%)` = ~ (length(.x[!is.na(.x)]) / length(.x))*100
        ),
    factor = sfl(
        `Data completeness (%)` = ~ (length(.x[!is.na(.x)]) / length(.x))*100,
    pct = ~{
        prt <- prop.table(table(., useNA = "no"))*100
        val <- sprintf("%.3f", prt)
        nm1 <- tolower(substr(names(prt), 1, 3))
            stringr::str_c(nm1, val, sep = ": ", collapse = ", ")
        }
        )
)
